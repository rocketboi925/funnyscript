local function getCharacter()
	local player = Players.LocalPlayer
	return player.Character or player.CharacterAdded:Wait()
end

local function getHRP()
	local char = getCharacter()
	return char and char:FindFirstChild("HumanoidRootPart")
end

local function teleportTo(position)
	local hrp = getHRP()
	if hrp then
		hrp.CFrame = CFrame.new(position + Vector3.new(0, 3, 0))
	end
end

local function waitForPickup(toolName, timeout)
	local player = Players.LocalPlayer
	local t = tick()
	while tick() - t < timeout do
		for _, item in ipairs(player.Backpack:GetChildren()) do
			if item.Name == toolName then return true end
		end
		local char = player.Character
		if char then
			for _, item in ipairs(char:GetChildren()) do
				if item.Name == toolName then return true end
			end
		end
		task.wait(0.1)
	end
	return false
end

local function getSelectedItemsByPriority()
	local selected = Options.FarmItems.Value
	local result = {}
	for _, itemName in ipairs(ITEM_PRIORITY) do
		if selected[itemName] then
			table.insert(result, itemName)
		end
	end
	return result
end

local function findItemsInWorkspace(priorityList)
	local found = {}

	local spawnedItems = workspace:FindFirstChild("ServerJunk") and workspace.ServerJunk:FindFirstChild("SpawnedItems")
	if not spawnedItems then return found end

	local priorityMap = {}
	for i, name in ipairs(priorityList) do
		priorityMap[name] = i
	end

	for _, item in ipairs(spawnedItems:GetChildren()) do
		local p = priorityMap[item.Name]
		if p then
			table.insert(found, { priority = p, instance = item })
		end
	end

	table.sort(found, function(a, b) return a.priority < b.priority end)

	return found
end

task.spawn(function()
	while task.wait(0.1) do
		if not Toggles.FarmToggle.Value then
			farmRunning = false
			task.wait(0.5)
		else
			farmRunning = true

			local priorityList = getSelectedItemsByPriority()
			if #priorityList == 0 then task.wait(1) continue end

			local items = findItemsInWorkspace(priorityList)

			if #items == 0 then
				if savedPosition then
					local hrp = getHRP()
					if hrp then
						local distFromSaved = (hrp.CFrame.Position - savedPosition).Magnitude
						if distFromSaved > 5 then
							teleportTo(savedPosition)
						end
					end
				end
				task.wait(1)
				continue
			end

			for _, entry in ipairs(items) do
				if not Toggles.FarmToggle.Value then break end

				local item = entry.instance
				if not item or not item.Parent then continue end

				local itemPos
				if item:IsA("BasePart") then
					itemPos = item.Position
				elseif item.PrimaryPart then
					itemPos = item.PrimaryPart.Position
				else
					local part = item:FindFirstChildWhichIsA("BasePart")
					if part then itemPos = part.Position end
				end

				if not itemPos then continue end

				teleportTo(itemPos)
				waitForPickup(item.Name, 2)
				task.wait(0.15)
			end

			task.wait(0.3)
		end
	end
end)

			Library:Notify("Farm stopped. Teleported back to saved position.")
			savedPosition = nil
		else
			Library:Notify("Farm stopped.")
		end
	end
end)

-- Weapon dropdown (reads from backpack)
local function getBackpackWeapons()
    local names = {}
    local p = Players.LocalPlayer
    for _, tool in ipairs(p.Backpack:GetChildren()) do
        if tool:IsA("Tool") then table.insert(names, tool.Name) end
    end
    local c = p.Character
    if c then
        for _, tool in ipairs(c:GetChildren()) do
            if tool:IsA("Tool") then
                local found = false
                for _, n in ipairs(names) do if n == tool.Name then found = true break end end
                if not found then table.insert(names, tool.Name) end
            end
        end
    end
    if #names == 0 then names = {"(No tools found)"} end
    table.sort(names)
    return names
end

local function hasMaskInInventory()
    local p = Players.LocalPlayer
    for _, tool in ipairs(p.Backpack:GetChildren()) do
        if tool.Name == "Stone Mask" then return true end
    end
    local char = p.Character
    if char then
        for _, tool in ipairs(char:GetChildren()) do
            if tool:IsA("Tool") and tool.Name == "Stone Mask" then return true end
        end
    end
    return false
end

local function hasMaskEquipped()
    local char = Players.LocalPlayer.Character
    if not char then return false end
    local tool = char:FindFirstChildOfClass("Tool")
    return tool and tool.Name == "Stone Mask"
end

local function findMaskInWorkspace()
    local spawnedItems = workspace:FindFirstChild("ServerJunk") and workspace.ServerJunk:FindFirstChild("SpawnedItems")
    if not spawnedItems then return nil end
    for _, item in ipairs(spawnedItems:GetChildren()) do
        if item.Name == "Stone Mask" then return item end
    end
    return nil
end

local function getMaskPosition(item)
    if not item or not item.Parent then return nil end
    local handle = item:FindFirstChild("Handle")
    if handle and handle:IsA("BasePart") then return handle.Position end
    if item:IsA("BasePart") then return item.Position end
    if item.PrimaryPart then return item.PrimaryPart.Position end
    local part = item:FindFirstChildWhichIsA("BasePart")
    if part then return part.Position end
    return nil
end

local function getSafeHRP()
    -- Never yields — safe to call inside pickup loops
    local char = Players.LocalPlayer.Character
    if not char then return nil end
    return char:FindFirstChild("HumanoidRootPart")
end



local function equipTool(toolName)
    local p = Players.LocalPlayer
    local char = p.Character
    if not char then return false end
    local already = char:FindFirstChildOfClass("Tool")
    if already and already.Name == toolName then return true end
    local tool = p.Backpack:FindFirstChild(toolName)
    if tool then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then hum:EquipTool(tool) return true end
    end
    return false
end

local function doClick()
    VIM:SendMouseButtonEvent(200, 200, 0, true, game, 1)
    VIM:SendMouseButtonEvent(200, 200, 0, false, game, 1)
end

local function isVampire()
    local data = game.Players.LocalPlayer:FindFirstChild("Data")
    if not data then return false end
    local unlocked = data:FindFirstChild("Unlocked")
    if not unlocked then return false end
    local val = unlocked:FindFirstChild("Vampire")
    local returnvalue = val.Value
    if returnvalue == "true" then
    return true
    elseif returnvalue == "false" then
        return false
    end
end

Toggles.PlayerFarmMain:OnChanged(function()
    if Toggles.PlayerFarmMain.Value then
        local char = Players.LocalPlayer.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then
            Library:Notify("No character found!")
            return
        end

        playerFarmConnection = RunService.Heartbeat:Connect(function()
            if not Toggles.PlayerFarmMain.Value then return end

            local p = Players.LocalPlayer
            local c = p.Character
            if not c then return end

            local h = c:FindFirstChild("HumanoidRootPart")
            if h then
                h.CFrame = CFrame.new(FARM_POSITION)
            end

               local mode = Options.MainFarmMode.Value



            if mode == "Vampire Kills" then

                -- Check for KO'd player within 3 studs

                local koTarget = nil

                local aliveFolder = workspace:FindFirstChild("Alive")

                if aliveFolder and h then

                    for _, playerFolder in ipairs(aliveFolder:GetChildren()) do

                        local status = playerFolder:FindFirstChild("Status")

                        if status then

                           local ko = status:FindFirstChild("KO")

if ko then

                                local targetHRP = playerFolder:FindFirstChildWhichIsA("BasePart")

                                    or (playerFolder.PrimaryPart)

                                -- Try to find their HRP more specifically

                                local found = playerFolder:FindFirstChild("HumanoidRootPart")

                                if found then

                                    local dist = (found.Position - h.Position).Magnitude

                                    if dist <= 3 then

                                        koTarget = playerFolder

                                        break

                                    end

                                end

                            end

                        end

                    end

                end



                if koTarget then

                    -- Prefer Absorption, fall back to Drain

                    local preferredTool = p.Backpack:FindFirstChild("Absorption") or p.Backpack:FindFirstChild("Drain")

                    local preferredName = preferredTool and preferredTool.Name or "Drain"

                    local equipped = c:FindFirstChildOfClass("Tool")



                    if not equipped or equipped.Name ~= preferredName then

                        -- Not equipped yet, equip it first and wait

                        if preferredTool then

                            local hum = c:FindFirstChildOfClass("Humanoid")

                            if hum then hum:EquipTool(preferredTool) end

                        end

                        return

                    end



                    -- Tool is equipped, click every heartbeat

                    VIM:SendMouseButtonEvent(200, 200, 0, true, game, 1)

                    task.wait(0.1)

                    VIM:SendMouseButtonEvent(200, 200, 0, false, game, 1)

                    task.wait(CLICK_DELAY)

                else

                    -- No KO target nearby: use selected weapon and punch normally

                    local selectedWeapon = Options.PlayerFarmWeapon.Value

                    if not selectedWeapon or selectedWeapon == "(No tools found)" then return end



                    local equipped = c:FindFirstChildOfClass("Tool")

                    if not equipped or equipped.Name ~= selectedWeapon then

                        local tool = p.Backpack:FindFirstChild(selectedWeapon)

                        if tool then

                            local hum = c:FindFirstChildOfClass("Humanoid")

                            if hum then hum:EquipTool(tool) end

                        end

                        return

                    end



                    local now = tick()

                    if now - lastClickTime >= CLICK_DELAY then

                        lastClickTime = now

                        doClick()

                    end

                end




            else
                -- Regular mode: equip selected weapon and spam click
                local selectedWeapon = Options.PlayerFarmWeapon.Value
                if not selectedWeapon or selectedWeapon == "(No tools found)" then return end

                local equipped = c:FindFirstChildOfClass("Tool")
                if not equipped or equipped.Name ~= selectedWeapon then
                    local tool = p.Backpack:FindFirstChild(selectedWeapon)
                    if tool then
                        local hum = c:FindFirstChildOfClass("Humanoid")
                        if hum then hum:EquipTool(tool) end
                    end
                    return
                end

                local now = tick()
                if now - lastClickTime >= CLICK_DELAY then
                    lastClickTime = now
                    doClick()
                end
            end
        end)

        Library:Notify("Player Farm started!")
    else
        if playerFarmConnection then
            playerFarmConnection:Disconnect()
            playerFarmConnection = nil
        end
        Library:Notify("Player Farm stopped.")
    end
end)

local camera = workspace.CurrentCamera

local NoSlowState = {
	currentVelocity = Vector3.new(0, 0, 0),
	wasSlowed = false,
	connection = nil,
}

local function StartNoSlow()
	if NoSlowState.connection then
		NoSlowState.connection:Disconnect()
	end
	NoSlowState.connection = RunService.Heartbeat:Connect(function()
		if not Toggles.NoSlowToggle or not Toggles.NoSlowToggle.Value then
			NoSlowState.wasSlowed = false
			return
		end
		local char = LocalPlayer.Character
		if not char then return end
		local hrp = char:FindFirstChild("HumanoidRootPart")
		local humanoid = char:FindFirstChildOfClass("Humanoid")
		if not hrp or not hrp.Parent or not humanoid or humanoid.Health <= 0 then return end

		if humanoid.WalkSpeed <= 8 and not NoSlowState.wasSlowed then
			NoSlowState.wasSlowed = true
		end

		local moveDir = Vector3.new(0, 0, 0)
		if UIS:IsKeyDown(Enum.KeyCode.W) then
			moveDir = moveDir + (camera.CFrame.LookVector * Vector3.new(1, 0, 1)).Unit
		end
		if UIS:IsKeyDown(Enum.KeyCode.A) then
			moveDir = moveDir - (camera.CFrame.RightVector * Vector3.new(1, 0, 1)).Unit
		end
		if UIS:IsKeyDown(Enum.KeyCode.S) then
			moveDir = moveDir - (camera.CFrame.LookVector * Vector3.new(1, 0, 1)).Unit
		end
		if UIS:IsKeyDown(Enum.KeyCode.D) then
			moveDir = moveDir + (camera.CFrame.RightVector * Vector3.new(1, 0, 1)).Unit
		end

		if moveDir.Magnitude > 0 then
			moveDir = moveDir.Unit
			NoSlowState.currentVelocity = moveDir * 16
		else
			NoSlowState.currentVelocity = NoSlowState.currentVelocity * Options.NoSlowFriction.Value
		end

		if NoSlowState.wasSlowed and humanoid.WalkSpeed <= 8 then
			local verticalVel = hrp.AssemblyLinearVelocity.Y
			hrp.AssemblyLinearVelocity = Vector3.new(
				NoSlowState.currentVelocity.X,
				verticalVel,
				NoSlowState.currentVelocity.Z
			)
		end

		if NoSlowState.wasSlowed and humanoid.WalkSpeed > 8 then
			NoSlowState.wasSlowed = false
		end
	end)
end

local function StopNoSlow()
	if NoSlowState.connection then
		NoSlowState.connection:Disconnect()
		NoSlowState.connection = nil
	end
	NoSlowState.wasSlowed = false
	NoSlowState.currentVelocity = Vector3.new(0, 0, 0)
end

local DATA_FOLDERS = { "EXP", "Unlocked" }

local enabledFields = {}
local fieldToggles = {}

local function scanDataFields()
	local fields = { EXP = {}, Unlocked = {} }

	local samplePlayer = LocalPlayer
	if not samplePlayer then
		for _, p in ipairs(Players:GetPlayers()) do
			samplePlayer = p
			break
		end
	end

	if not samplePlayer then return fields end

	local data = samplePlayer:FindFirstChild("Data")
	if not data then return fields end

	for _, folderName in ipairs(DATA_FOLDERS) do
		local folder = data:FindFirstChild(folderName)
		if folder then
			for _, child in ipairs(folder:GetChildren()) do
				if child:IsA("IntValue") or child:IsA("StringValue") or child:IsA("NumberValue") or child:IsA("BoolValue") then
					table.insert(fields[folderName], child.Name)
				end
			end
		end
	end

	return fields
end

local function buildFieldToggles()
	local fields = scanDataFields()

	for _, folderName in ipairs(DATA_FOLDERS) do
		if #fields[folderName] > 0 then
			ESPDataGroup:AddLabel("── " .. folderName .. " ──", false)
			for _, fieldName in ipairs(fields[folderName]) do
				local idx = "ESPField_" .. folderName .. "_" .. fieldName
				ESPDataGroup:AddToggle(idx, {
					Text = fieldName,
					Default = false,
					Tooltip = "Show " .. folderName .. "." .. fieldName .. " in ESP",
				})
				enabledFields[idx] = { folder = folderName, field = fieldName }
				table.insert(fieldToggles, idx)
			end
		end
	end
end

task.spawn(function()
	task.wait(3)
	buildFieldToggles()
end)

local espBillboards = {}

local function getFieldValue(player, folderName, fieldName)
	local data = player:FindFirstChild("Data")
	if not data then return nil end
	local folder = data:FindFirstChild(folderName)
	if not folder then return nil end
	local val = folder:FindFirstChild(fieldName)
	if not val then return nil end
	return val.Value
end

local function getPlayerRace(player)
	local data = player:FindFirstChild("Data")
	if not data then return "Unknown" end
	local unlocked = data:FindFirstChild("Unlocked")
	if not unlocked then return "Human" end

	local races = { "Vampire", "Hamon", "Pillarmen", "Zombie" }
	for _, race in ipairs(races) do
		local val = unlocked:FindFirstChild(race)
		if val and (val.Value == true or val.Value == "true") then
			return race
		end
	end
	return "Human"
end

local function buildESPText(player)
	local lines = {}

	table.insert(lines, player.DisplayName .. " (@" .. player.Name .. ")")

	if Toggles.ESPShowHealth and Toggles.ESPShowHealth.Value then
		local char = player.Character
		local hum = char and char:FindFirstChildOfClass("Humanoid")
		if hum then
			table.insert(lines, "❤️ " .. math.floor(hum.Health) .. " / " .. math.floor(hum.MaxHealth))
		end
	end

	if Toggles.ESPShowRace and Toggles.ESPShowRace.Value then
		table.insert(lines, "🧬 " .. getPlayerRace(player))
	end

	if Toggles.ESPShowDistance.Value then
		local char = player.Character
		local localChar = LocalPlayer.Character
		if char and localChar then
			local hrp = char:FindFirstChild("HumanoidRootPart")
			local lhrp = localChar:FindFirstChild("HumanoidRootPart")
			if hrp and lhrp then
				local dist = math.floor((hrp.Position - lhrp.Position).Magnitude)
				table.insert(lines, "📍 " .. dist .. " studs")
			end
		end
	end

	for _, idx in ipairs(fieldToggles) do
		if Toggles[idx] and Toggles[idx].Value then
			local info = enabledFields[idx]
			local val = getFieldValue(player, info.folder, info.field)
			if val ~= nil then
				table.insert(lines, info.field .. ": " .. tostring(val))
			end
		end
	end

	return table.concat(lines, "\n")
end

local function refreshAllESP()
	if not Toggles.ESPEnabled.Value then
		for player, _ in pairs(espBillboards) do
			removeESP(player)
		end
		return
	end

	if not Toggles.ESPPlayers.Value then
		for player, _ in pairs(espBillboards) do
			removeESP(player)
		end
		return
	end

	for _, player in ipairs(Players:GetPlayers()) do
		if player == LocalPlayer and not Toggles.ESPShowSelf.Value then
			removeESP(player)
			continue
		end

		if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
			if not espBillboards[player] then
				createBillboard(player)
			end
		end
	end
end

		if not Toggles.ESPEnabled then continue end
		if not Toggles.ESPEnabled.Value then continue end

		if not Toggles.ESPPlayers.Value then
			for player, _ in pairs(espBillboards) do
				removeESP(player)
			end
			continue
		end

		for _, player in ipairs(Players:GetPlayers()) do
			if player == LocalPlayer and not Toggles.ESPShowSelf.Value then continue end

			local bb = espBillboards[player]
			if not bb then
				if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
					createBillboard(player)
					bb = espBillboards[player]
				end
			end

			if bb then
				local label = bb:FindFirstChild("Frame") and bb.Frame:FindFirstChild("ESPLabel")
				if label then

local npcESPBillboards = {}

local function getNPCRootPart(npcModel)
	local hrp = npcModel:FindFirstChild("HumanoidRootPart")
	if hrp then return hrp end
	return npcModel.PrimaryPart or npcModel:FindFirstChildWhichIsA("BasePart")
end

local function createNPCBillboard(npcModel, labelColor)
	local root = getNPCRootPart(npcModel)
	if not root then return end

	if npcESPBillboards[npcModel] then
		npcESPBillboards[npcModel]:Destroy()
		npcESPBillboards[npcModel] = nil
	end

	local bb = Instance.new("BillboardGui")
	bb.Name = "HamonNPCESP"
	bb.Adornee = root
	bb.AlwaysOnTop = true
	bb.StudsOffset = Vector3.new(0, 3.5, 0)
	bb.Size = UDim2.new(0, Options.ESPBoxWidth.Value, 0, 60)
	bb.MaxDistance = Options.ESPMaxDistance.Value
	bb.ResetOnSpawn = false
	bb.Parent = game:GetService("CoreGui")

	local bg = Instance.new("Frame")
	bg.Size = UDim2.new(1, 0, 1, 0)
	bg.BackgroundColor3 = Options.ESPBGColor.Value
	bg.BackgroundTransparency = Options.ESPBGTransparency.Value
	bg.BorderSizePixel = 0
	bg.Parent = bb

	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, 6)
	corner.Parent = bg

	local padding = Instance.new("UIPadding")
	padding.PaddingLeft = UDim.new(0, 5)
	padding.PaddingRight = UDim.new(0, 5)
	padding.PaddingTop = UDim.new(0, 4)
	padding.PaddingBottom = UDim.new(0, 4)
	padding.Parent = bg

	local label = Instance.new("TextLabel")
	label.Name = "ESPLabel"
	label.Size = UDim2.new(1, 0, 1, 0)
	label.BackgroundTransparency = 1
	label.TextColor3 = labelColor
	label.TextSize = Options.ESPTextSize.Value
	label.Font = Enum.Font.GothamBold
	label.TextXAlignment = Enum.TextXAlignment.Left
	label.TextYAlignment = Enum.TextYAlignment.Top
	label.TextWrapped = true
	label.Text = "[NPC] " .. npcModel.Name
	label.Parent = bg

	npcESPBillboards[npcModel] = bb
end

local function removeNPCBillboard(npcModel)
	if npcESPBillboards[npcModel] then
		npcESPBillboards[npcModel]:Destroy()
		npcESPBillboards[npcModel] = nil
	end
end

task.spawn(function()
	while task.wait(0.15) do
		local masterOn = Toggles.ESPEnabled and Toggles.ESPEnabled.Value
		local humanOn = masterOn and Toggles.ESPHumanNPC and Toggles.ESPHumanNPC.Value
		local zombieOn = masterOn and Toggles.ESPZombieNPC and Toggles.ESPZombieNPC.Value

		-- Clean up billboards for NPCs no longer in workspace
		for model, _ in pairs(npcESPBillboards) do
			if not model or not model.Parent then
				removeNPCBillboard(model)
			end
		end

		if not humanOn and not zombieOn then
			for model, _ in pairs(npcESPBillboards) do
				removeNPCBillboard(model)
			end
			continue
		end

		local aliveFolder = workspace:FindFirstChild("Alive")
		if not aliveFolder then continue end

		local localHRP = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")

		for _, child in ipairs(aliveFolder:GetChildren()) do
			local name = child.Name
			local isHuman = string.sub(name, 1, 5) == "Human"
			local isZombie = string.sub(name, 1, 6) == "Zombie"

			local shouldShow = (isHuman and humanOn) or (isZombie and zombieOn)

			if not shouldShow then
				removeNPCBillboard(child)
				continue
			end

			local root = getNPCRootPart(child)
			if not root then continue end

			if not npcESPBillboards[child] then
				local color = isZombie and Color3.fromRGB(100, 255, 100) or Color3.fromRGB(255, 200, 80)
				createNPCBillboard(child, color)
			end

			local bb = npcESPBillboards[child]
			if bb then
				local lbl = bb:FindFirstChild("Frame") and bb.Frame:FindFirstChild("ESPLabel")
				if lbl then
					local distStr = ""
					if localHRP then
						local dist = math.floor((root.Position - localHRP.Position).Magnitude)
						distStr = "\n📍 " .. dist .. " studs"
					end
					lbl.Text = "[NPC] " .. name .. distStr
					lbl.TextSize = Options.ESPTextSize.Value
				end
				local bgFrame = bb:FindFirstChild("Frame")
				if bgFrame then
					bgFrame.BackgroundColor3 = Options.ESPBGColor.Value
					bgFrame.BackgroundTransparency = Options.ESPBGTransparency.Value
				end
				bb.MaxDistance = Options.ESPMaxDistance.Value
				bb.Size = UDim2.new(0, Options.ESPBoxWidth.Value, 0, 60)
			end
		end
	end
end)

local function getShopItems()
	local items = {}

					hasClickDetector = true
				end

		end
	end

local function buildTargetStats(target)
	if not target then return "No target selected" end

	local data = target:FindFirstChild("Data")
	if not data then return "No data found for " .. target.Name end

	local lines = {}
	table.insert(lines, "=== " .. target.Name .. " ===")

	for _, folder in ipairs(data:GetChildren()) do
		if folder:IsA("Folder") then
			table.insert(lines, "[ " .. folder.Name .. " ]")
			for _, val in ipairs(folder:GetChildren()) do
				if val:IsA("IntValue") or val:IsA("StringValue") or val:IsA("NumberValue") or val:IsA("BoolValue") then
					table.insert(lines, "  " .. val.Name .. ": " .. tostring(val.Value))
				end
			end
		elseif folder:IsA("IntValue") or folder:IsA("StringValue") or folder:IsA("NumberValue") or folder:IsA("BoolValue") then
			table.insert(lines, folder.Name .. ": " .. tostring(folder.Value))
		end
	end

	if #lines == 1 then
		table.insert(lines, "No readable values found")
	end

	return table.concat(lines, "\n")
end

local S_H = game:GetService("HttpService")
local S_T = game:GetService("TeleportService")

local function fetchServers(sortLowest)
	local servers = {}
	local cursor = ""

	-- Fetch up to 2 pages for a decent pool
	for _ = 1, 2 do
		local url = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
		if cursor ~= "" then
			url = url .. "&cursor=" .. cursor
		end

		local success, result = pcall(function()
			return game:HttpGet(url)
		end)
		if not success then break end

		local parsed = pcall(function()
			parsed = S_H:JSONDecode(result)
		end) and parsed or S_H:JSONDecode(result)

		if not parsed or not parsed.data then break end

		for _, v in ipairs(parsed.data) do
			if tostring(v.id) ~= tostring(game.JobId) and v.playing and v.maxPlayers and v.playing < v.maxPlayers then
				table.insert(servers, v)
			end
		end

		if parsed.nextPageCursor and parsed.nextPageCursor ~= "null" and parsed.nextPageCursor ~= nil then
			cursor = parsed.nextPageCursor
		else
			break
		end
	end

	return servers
end

ExtrasGroup:AddButton("Hop to Lowest Population", function()
	Library:Notify("Fetching servers...")
	task.spawn(function()
		local ok, servers = pcall(fetchServers, true)
		if not ok or #servers == 0 then
			Library:Notify("No servers found!")
			return
		end
		table.sort(servers, function(a, b) return a.playing < b.playing end)
		local target = servers[1]
		Library:Notify("Joining server with " .. target.playing .. " player(s)...")
		pcall(function()
			S_T:TeleportToPlaceInstance(game.PlaceId, target.id, Players.LocalPlayer)
		end)
	end)
end)
